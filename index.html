Questions related to inheritance & prototype

1.Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype 
        parentObj = {
            calAge: function() {
                console.log(`${this.name} ${this.age}`)
            }
        }
        
        let childObj = Object.create(parentObj);
        
        childObj.name = 'Anamika';
        childObj.age = 12;
        
        childObj.calAge();
2.Write code to explain prototype chaining
        let grandfather = {
            age: 70,
            location: 'Hyderabad'
        }
        
        const father = Object.create(grandfather);
        
        father.name = 'John';
        
        console.log(father.name);
        console.log(father.age);  // father object don't have prototype age but it can access grandfather object prototype beacuse of prototype chaining
3.Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays
        parentObj = {
            sum: function(arr) {
                let sum = 0;
                for(let i = 0;i< arr.length;i++) {
                    sum += arr[i];
                }
                console.log(sum);
            }
        }
        
        let childObj = Object.create(parentObj);
        childObj.sum([1, 2, 3, 4, 5]);
        childObj.sum([3, 4, 5, 6, 7])
